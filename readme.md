#### 1 使用 npm init 在命令行 生成一个package.json 文件

```json
{
  "name": "test1-nodejs-backend", // 项目名称
  "version": "1.0.0", // 版本号
  "description": "---", // 当前项目的描述
  "main": "index.js", //
  "scripts": { // 执行脚本
    "start": "node index.js"
  },
  "dependencies": { // 依赖
    "@types/node": "latest"
  },
  "author": "", // 作者
  "license": "ISC" //
}

```
